— 메시지 로그 —

postMessage 수신 이벤트

// 부모 페이지에서 수신하는 이벤트
window.addEventListener('message', (e) => {
  switch (e.data.type) {
    case 'POA_READY':
      console.log('에디터 준비 완료', e.data.version);
      break;
    case 'POA_CONTENT_CHANGED':
      console.log('내용 변경:', e.data.charCount, '자', e.data.wordCount, '단어');
      break;
    case 'POA_CONTENT':
      console.log('에디터 HTML:', e.data.html);
      break;
    case 'POA_SAVED':
      console.log('저장 완료:', e.data.docKey, e.data.title);
      break;
    case 'POA_FULLSCREEN_REQUEST':
      // iframe에서 전체화면 요청
      iframe.requestFullscreen?.();
      break;
  }
});

postMessage 발송 명령

const iframe = document.querySelector('iframe');

// 크기 변경
iframe.contentWindow.postMessage({ type: 'POA_RESIZE', height: '800px' }, '*');

// 전체화면 진입
iframe.contentWindow.postMessage({ type: 'POA_FULLSCREEN', enabled: true }, '*');

// HTML 가져오기 (POA_CONTENT 이벤트로 응답)
iframe.contentWindow.postMessage({ type: 'POA_GET_CONTENT' }, '*');

// HTML 설정
iframe.contentWindow.postMessage({ type: 'POA_SET_CONTENT', html: '<p>내용</p>' }, '*');
